We can use square brackets in an object literal,
when creating an object.That's called computed properties. 

        code 1:
            let fruit =prompt("Which fruit to buy ?","apple");
    
            let bag={
                [fruit]:5//the name of the property is taken from the variable fruit
            };
            alert(bag.apple);
            
        
    
        Code 2:
            //same code concept of above
            let fruits =prompt("which fruits to buy ?","apple");//we can change fruits instead of fruit because multi redeclaration is not allow in let
             
            let bags={};
            //take property name from the fruit variable
            bags[fruits]=5;
            alert(bags.apple);//5
            alert(bags.b);//undefined
    

        code 3:
          //concatenated the variable and string in square bracket in object of key
            let toy="car";
    
            let box={
                [toy+'key']:2//box.carkey=2
            };
            alert(box.carkey);//2
